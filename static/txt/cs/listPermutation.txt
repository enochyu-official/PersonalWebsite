#include <iostream>
#include <vector>
#include <algorithm>

int n;
int count = 0;
std::vector<int> arr;

void defArr() {
    for (int i = 1; i <= n; i++) {
        arr.push_back(i);
    }
}

void showArr() {
    for (int i = 0; i < n; i++) {
        std::cout << arr[i];
    }
    std::cout << std::endl;
}

void permute(int a, int b) {
    if (a == b) {
        showArr();
        count++;
    }

    for (int i = a; i <= b; i++) {
        std::swap(arr[a], arr[i]);
        permute(a + 1, b);
        std::swap(arr[a], arr[i]);
    }
}


int main() {
    std::cin >> n;
    defArr();
    permute(0, n - 1);
    std::cout << count << std::endl;

    return 0;
}
